    \[
        \dgM(F, G)_i = \dgM(F, \Sigma^i G)_0

    By \autoref{lem:dgmod_shift_eq_plus},

        \dgM(F, G)_i = \dgM(\Sigma^{-i} F, G)_0.
    \]
    Then, by applying \( \Sigma^i \), we get the result.
\end{proof}
