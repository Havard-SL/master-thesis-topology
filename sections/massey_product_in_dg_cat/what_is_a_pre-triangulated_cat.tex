        }

            \[ 
                \tuple{\Hom_{C_{dg}(\Mod(R))}(A, B)}_i := \bigoplus_{j \in \Zb} \Hom_{\Mod(R)}(A_j, B_{j + i})

            \[
                \Hom_{C_{dg}(\Mod(R))}(A, B) = \bigoplus_{i \in \Zb} \tuple{\Hom_{C_{dg}(\Mod(R))}(A, B)}_i.
